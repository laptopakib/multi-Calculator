let inputView = document.getElementById('input-view'); let resultView = document.getElementById('result-view'); let modeLabel = document.getElementById('mode-label'); let expression = ""; // ABC Panel Setup const abcPanel = document.getElementById('alphabet-panel'); const letters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".split(""); letters.forEach(l => { let btn = document.createElement("button"); btn.innerText = l; btn.className = "abc-key"; btn.onclick = () => { add(l); toggleABC(); }; abcPanel.appendChild(btn); }); function add(val) { expression += val; inputView.innerText = expression; } function addConst(c) { const vals = { 'G': 6.674e-11, 'g': 9.806, 'R': 8.314, 'F': 96485 }; expression += vals[c]; inputView.innerText = expression; } function toggleABC() { let pnl = document.getElementById('alphabet-panel'); pnl.style.display = (pnl.style.display === 'grid') ? 'none' : 'grid'; modeLabel.innerText = "ALPHABET MODE"; } function switchMode(mode) { modeLabel.innerText = mode + " MODE"; if(mode === 'GEO') { alert("Geometry Mode: Enter Area/Volume formula or values."); expression = "Math.PI * "; // জ্যামিতির বেস inputView.innerText = expression; } } function solve() { try { let tempExp = expression; // Algebra Solver: অক্ষর থাকলে মান চাইবে if (/[A-Z]/.test(tempExp)) { let variables = tempExp.match(/[A-Z]/g); variables.forEach(v => { let val = prompt(`Enter value for ${v}:`); tempExp = tempExp.replace(new RegExp(v, 'g'), val); }); } // ব্র্যাকেট অটো ক্লোজ লজিক let open = (tempExp.match(/\(/g) || []).length; let close = (tempExp.match(/\)/g) || []).length; while(open > close) { tempExp += ")"; close++; } let res = eval(tempExp.replace(/×/g, '*').replace(/÷/g, '/')); resultView.innerText = Number.isInteger(res) ? res : res.toFixed(6); } catch (e) { resultView.innerText = "Error"; } } function openConverter() { window.location.href = "converter.html"; // কনভার্টার পেজে নিয়ে যাবে } function cls() { expression = ""; inputView.innerText = ""; resultView.innerText = "0"; } function del() { expression = expression.slice(0, -1); inputView.innerText = expression; }
